Audio Player Logic // Data for players const songs = [ { id: 1, title: "Wedding Song - Anjali & Raj", duration: "2:28" }, { id: 2, title: "Birthday Surprise - Rohan", duration: "2:37" }, { id: 3, title: "Romantic Dedication - Sneha", duration: "2:56" }, { id: 4, title: "Lofi Chill - Study Time", duration: "2:23" }, { id: 5, title: "Anniversary Special - Priya & Aman", duration: "3:10" }, { id: 6, title: "Romantic Duet - Deepak & Sakshi", duration: "3:41" } ]; const audioGrid = document.getElementById('audio-grid'); // Function to create HTML for a player function createPlayerHTML(song) { return `

${song.title}

0:00 / ${song.duration}
`; } // Render players audioGrid.innerHTML = songs.map(createPlayerHTML).join(''); // Re-init icons for the newly added HTML lucide.createIcons(); // Add Logic to Players document.querySelectorAll('.audio-player').forEach(player => { const btn = player.querySelector('.play-btn'); const playIcon = player.querySelector('.icon-play'); const pauseIcon = player.querySelector('.icon-pause'); const progressBar = player.querySelector('.progress-bar'); let isPlaying = false; let progress = 0; let interval; btn.addEventListener('click', () => { isPlaying = !isPlaying; if (isPlaying) { // Switch Icons playIcon.classList.add('hidden'); pauseIcon.classList.remove('hidden'); // Start Interval interval = setInterval(() => { if (progress >= 100) progress = 0; progress += 1; progressBar.style.width = `${progress}%`; }, 100); } else { // Switch Icons playIcon.classList.remove('hidden'); pauseIcon.classList.add('hidden'); // Clear Interval clearInterval(interval); } }); });
